## Install

```bash
npm install --save react-use-mapped-state
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'react-use-mapped-state'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
